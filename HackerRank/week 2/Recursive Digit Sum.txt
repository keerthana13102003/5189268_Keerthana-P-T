#include<stdio.h>
#include<string.h>

long long sd(long long x){
    if(x<10)return x;
    long long s=0;
    while(x){s+=x%10;x/=10;}
    return sd(s);
}

int main(){
    char n[100005];long long k;
    scanf("%s %lld",n,&k);
    long long s=0;
    for(int i=0;n[i];i++)s+=n[i]-'0';
    s*=k;
    printf("%lld\n",sd(s));
    return 0;
}
