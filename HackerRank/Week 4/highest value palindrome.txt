#include <stdio.h>
#include <stdlib.h>
#include <string.h>

char*highestValuePalindrome(char*s,int n,int k){
    int i=0,u=n-1,y=0;
    int*t=(int*)calloc(n,sizeof(int));
    while(i<u){
        if(s[i]!=s[u]){
            if(k==0)return "-1";
            if(s[i]>s[u])s[u]=s[i];
            else s[i]=s[u];
            t[i]=t[u]=1;
            k--;
            y++;
        }
        i++;u--;
    }
    if(k<0)return "-1";
    i=0;u=n-1;
    while(i<=u&&k>0){
        if(i==u){
            if(s[i]!='9'&&k>0)s[i]='9';
        }else{
            if(s[i]!='9'){
                if(t[i]&&k>=1){
                    s[i]=s[u]='9';
                    k--;
                }else if(!t[i]&&k>=2){
                    s[i]=s[u]='9';
                    k-=2;
                }
            }
        }
        i++;u--;
    }
    return s;
}

int main(){
    int n,k;
    scanf("%d%d",&n,&k);
    char*s=(char*)malloc((n+1)*sizeof(char));
    scanf("%s",s);
    char*res=highestValuePalindrome(s,n,k);
    printf("%s\n",res);
    return 0;
}
