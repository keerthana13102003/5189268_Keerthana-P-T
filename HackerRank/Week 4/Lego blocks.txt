#include <stdio.h>
#include <stdlib.h>
#define MOD 1000000007

long long power(long long k,long long h){
    long long f=1;
    while(h>0){
        if(h%2==1)f=(f*k)%MOD;
        k=(k*k)%MOD;
        h=h/2;
    }
    return f;
}

int main(){
    int k;
    if(scanf("%d",&k)!=1) return 0;
    while(k--){
        int h,f;
        scanf("%d%d",&h,&f);

        long long *r=(long long*)malloc(sizeof(long long)*(f+1));
        long long *v=(long long*)malloc(sizeof(long long)*(f+1));
        long long *l=(long long*)malloc(sizeof(long long)*(f+1));

        r[0]=1;
        for(int i=1;i<=f;i++){
            r[i]=0;
            if(i-1>=0) r[i]=(r[i]+r[i-1])%MOD;
            if(i-2>=0) r[i]=(r[i]+r[i-2])%MOD;
            if(i-3>=0) r[i]=(r[i]+r[i-3])%MOD;
            if(i-4>=0) r[i]=(r[i]+r[i-4])%MOD;
        }

        for(int i=1;i<=f;i++) v[i]=power(r[i],h);

        l[1]=v[1];
        for(int i=2;i<=f;i++){
            long long sum=0;
            for(int u=1;u<i;u++){
                sum=(sum+l[u]*v[i-u])%MOD;
            }
            l[i]=(v[i]-sum+MOD)%MOD;
        }

        printf("%lld\n",l[f]);

        free(r);
        free(v);
        free(l);
    }
    return 0;
}
