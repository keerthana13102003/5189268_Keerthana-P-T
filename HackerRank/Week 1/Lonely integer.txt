#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);  
    int arr[100];
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);  
    }

    int unique = 0;
    for (int i = 0; i < n; i++) {
        unique ^= arr[i];  
    }

    printf("%d\n", unique); 
    return 0;
}
