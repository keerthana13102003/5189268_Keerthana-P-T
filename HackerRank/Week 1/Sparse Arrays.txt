#include <stdio.h>
#include <string.h>

int main() {
    int totalStrings;
    scanf("%d", &totalStrings);

    char data[1000][25]; // store all main strings
    for (int i=0; i<totalStrings; i++) {
        scanf("%s", data[i]);
    }

    int totalQueries;
    scanf("%d", &totalQueries);

    char checkList[1000][25]; // store all queries
    for (int i=0; i<totalQueries; i++) {
        scanf("%s", checkList[i]);
    }

    for (int i = 0; i < totalQueries; i++) {
        int matchCount = 0;
        for (int j=0; j<totalStrings; j++) {
            if (strcmp(checkList[i], data[j]) == 0) {
                matchCount++;
            }
        }
        printf("%d\n", matchCount);
    }

    return 0;
}
