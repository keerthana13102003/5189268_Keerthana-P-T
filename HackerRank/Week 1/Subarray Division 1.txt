// problem 11
#include<stdio.h>
int main(){
    int n;
    scanf("%d", &n);
    int s[n];
    for(int i=0; i<n; i++)
    scanf("%d", &s[i]);
    int d,m;
    scanf("%d %d",&d,&m);

    int cnt=0;
    for(int i=0;i<=n-m;i++) {
        int sum=0;
        for(int j=0;j<m;j++) {
            sum+=s[i+j];
        }
        if(sum==d) cnt++;
    }
    printf("%d\n",cnt);
    return 0;
}
