#include <stdio.h>
#include <stdlib.h>

int diagonalDifference(int n, int arr[n][n]) {
    int sum1 = 0, sum2 = 0;
    for (int i=0; i<n; i++) {
        sum1 += arr[i][i];
        sum2 += arr[i][n - 1 - i];
    }
    return abs(sum1 - sum2);
}

int main() {
    int n;
    scanf("%d", &n);
    int arr[n][n];
    for (int i=0; i<n; i++) {
        for (int j = 0; j < n; j++) {
            scanf("%d", &arr[i][j]);
        }
    }
    printf("%d\n", diagonalDifference(n, arr));
    return 0;
}
