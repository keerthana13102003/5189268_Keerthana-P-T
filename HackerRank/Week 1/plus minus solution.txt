#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);

    int arr[n];
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    int pos = 0, neg=0, zer= 0;
    for (int i = 0; i < n; i++) {
        if (arr[i] > 0) {
            pos++;
        } else if (arr[i] < 0) {
            neg++;
        } else {
            zer++;
        }
    }

    double total = n;
    printf("%.6f\n", pos / total);
    printf("%.6f\n", neg / total);
    printf("%.6f\n", zer / total);

    return 0;
}
