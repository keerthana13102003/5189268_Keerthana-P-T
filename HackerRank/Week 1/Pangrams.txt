#include <stdio.h>

int main() {
    char s[1001];
    int letters[26] = {0}, i;
    fgets(s, sizeof(s), stdin);
    for (i = 0; s[i] != '\0'; i++) {
        if (s[i] >= 'A' && s[i] <= 'Z') s[i] = s[i] + 32;
        if (s[i] >= 'a' && s[i] <= 'z') {
            letters[s[i] - 'a'] = 1;
        }
    }
    for (i=0; i<26; i++) {
        if (!letters[i]) {
            printf("not pangram");
            return 0;
        }
    }
    printf("pangram");
    return 0;
}
