#include <stdio.h>
#include<stdlib.h>
int main()
{
    int t;
    scanf("%d", &t);
    while(t--){
        int n,k;
        scanf("%d %d",&n,&k);
        int *a=(int*)malloc(n*sizeof(int));
        int *b=(int*)malloc(n*sizeof(int));
        for(int i=0;i<n;i++) scanf("%d",&a[i]);
        for(int i=0;i<n;i++) scanf("%d",&b[i]);
        for(int i=0;i<n-1;i++){
            for(int j=i+1;j<n;j++){
                if(a[i]>a[j]){
                    int tmp=a[i]; a[i]=a[j]; a[j]=tmp;
    }
    }
        }
        for(int i=0;i<n-1;i++){
            for(int j=i+1;j<n;j++){
                if(b[i]<b[j]){
                    int tmp=b[i]; b[i]=b[j]; b[j]=tmp;
                }
            }
        }
        int f=1;
        for(int i=0;i<n;i++){
            if(a[i]+b[i]<k){ f=0; break; }
        }
        if(f) printf("YES\n");
        else printf("NO\n");
        free(a);
        free(b);
    }
    return 0;
}
