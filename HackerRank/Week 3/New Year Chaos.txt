#include<stdio.h>
int k[200005];
int main(){
    int t,r,j,e,m;
    long long b;
    if(scanf("%d",&t)!=1)return 0;
    while(t--){
        scanf("%d",&r);
        for(j=0;j<r;j++)scanf("%d",&k[j]);
        m=0;
        for(j=0;j<r;j++){if(k[j]-(j+1)>2){m=1;break;}}
        if(m){puts("Too chaotic");continue;}
        b=0;
        for(j=0;j<r;j++){
            e=k[j]-2;
            if(e<0)e=0;
            for(m=e;m<j;m++)if(k[m]>k[j])b++;
        }
        printf("%lld\n",b);
    }
    return 0;
}
